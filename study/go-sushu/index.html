go-素数筛

十二月 17, 2022 · 1 分钟 · Ken | Suggest Changes

问题描述: 素数是一个自然数,它具有两个截然不同的自然数除数:1和它本身,


package main

func Generate(ch chan<- int) {
	for i := 2; ; i++ {
		ch <- i
	}
}

func Filter(in <-chan int, out chan<- int, prime int) {
	for {
		i := <-in
     
		if i%prime != 0 {
			out <- i
		}
	}
}

func main() {
	ch := make(chan int)
	go Generate(ch)
	for i := 0; i < 1; i++ {
		prime := <-ch

		print(prime, "\n")
		ch1 := make(chan int)
		go Filter(ch, ch1, prime)
		ch = ch1
	}
}
