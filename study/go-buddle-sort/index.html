go-冒泡排序算法

十二月 8, 2022 · 1 分钟 · Ken | Suggest Changes 
package main

import "fmt"

func main() {
	var arr = []int{9, 7, 6, 8, 4, 1, 5, 2, 3}
	for i := len(arr) - 1; i > 0; i-- {
		for j := 0; j < len(arr)-1; j++ {
			if arr[j] > arr[j+1] {
				swap(arr, j, j+1)
			}
		}
		fmt.Printf("arr: %v\n", arr)
	}

}
func swap(arr []int, a, b int) {
	var temp = arr[a]
	arr[a] = arr[b]
	arr[b] = temp
}
